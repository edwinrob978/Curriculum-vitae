Separador de vocales y consonantes 

                                
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.IO;

namespace Archivos
{
    class Program
    {
        static void Main(string[] args)
        {
            try {
                TextReader lee = new StreamReader("Prueba.txt");
                string datos = lee.ReadToEnd();
                lee.Close();
                TextWriter vocales;
                vocales = new StreamWriter("Vocales.txt");
                TextWriter consonantes;
                consonantes = new StreamWriter("Consonantes.txt");
                TextWriter union;
                union = new StreamWriter("Unir.txt");

                vocales.WriteLine(DejaVocal(datos).ToString());
                vocales.Close();

                consonantes.WriteLine(DejaConsonantes(datos).ToString());
                consonantes.Close();

                Console.WriteLine("Proceso Finalizado ....");
                Console.ReadKey();

                TextReader v = new StreamReader("Vocales.txt");
                TextReader c = new StreamReader("Consonantes.txt");

                String datos1 = v.ReadToEnd();
                String datos2 = c.ReadToEnd();

                v.Close();
                c.Close();

                union.WriteLine(UneDatos(datos1,datos2).ToString());
                union.Close();

            }
            catch (Exception ex)
            {
                Console.WriteLine(ex.Message);
            }
        }

        public static string DejaVocal(string cadena)
        {
            char[] datos = cadena.ToCharArray();
            string vocales = "";
            foreach (char v in datos)
            {
                if(v.ToString().ToLower().Equals("a") || v.ToString().ToLower().Equals("e") || v.ToString().ToLower().Equals("i") || v.ToString().ToLower().Equals("o") || v.ToString().ToLower().Equals("u"))
                {
                    vocales += v;
                }
                else
                {
                    vocales += " ";
                }
            }
            return vocales; 
        }

        public static string DejaConsonantes(string cadena)
        {
            char[] datos = cadena.ToCharArray();
            string consonantes = "";
            foreach (char c in datos)
            {
                if (!c.ToString().ToLower().Equals("a") && !c.ToString().ToLower().Equals("e") && !c.ToString().ToLower().Equals("i") && !c.ToString().ToLower().Equals("o") && !c.ToString().ToLower().Equals("u"))
                {
                    consonantes += c;
                } 
                else
                {
                    consonantes += " ";
                }
            }
            return consonantes;
        }

        public static string UneDatos(string cad1, string cad2)
        {
            char[] datos1 = cad1.ToCharArray();
            char[] datos2 = cad2.ToCharArray();
            string todo = "";
            int i = 0;

            foreach (char caracteres in datos1)
            {
                if (!caracteres.ToString().Equals(" "))
                {
                    todo += caracteres;
                    i++;
                }
                else
                {
                    todo += datos2[i];
                    i++;
                }
            }
            return todo;
        }
    }
}

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;
using System.Threading;



namespace Reproductor
{
    public partial class Form1 : Form
    {
        
        OpenFileDialog Abrir = new OpenFileDialog();

        int tiempo;

        public Form1()
        {
            InitializeComponent();
        }

       

        private void abrirToolStripMenuItem_Click(object sender, EventArgs e)
        {
            Abrir.Title = "Buscar Archivo";
            Abrir.FilterIndex = 1;

            try
            {
                if (Abrir.ShowDialog() == System.Windows.Forms.DialogResult.OK)
                {
                    axWindowsMediaPlayer1.URL = Abrir.FileName;
                    label1.Text = Abrir.FileName;
                }

            }
            catch (Exception)
            {
                return;
            }
           
            
            timer1.Interval = 1000;
            timer1.Start();           

        }

        private void timer1_Tick(object sender, EventArgs e)
        {

            progressBar1.Maximum = Convert.ToInt32(axWindowsMediaPlayer1.currentMedia.duration); // tiempo total del video 
            int actual = Convert.ToInt32(axWindowsMediaPlayer1.Ctlcontrols.currentPosition); //tiempo actual del video

            tiempo = progressBar1.Maximum;
            

             if (actual < progressBar1.Maximum)
             {
                for (int i = 0; i <= actual; i++)
                {
                    
                    progressBar1.Value = Convert.ToInt32(axWindowsMediaPlayer1.Ctlcontrols.currentPosition);

                }
               
             }
            else
            {
                timer1.Stop();
            }
             
        }

        private void button2_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.URL = Abrir.FileName;
            axWindowsMediaPlayer1.Ctlcontrols.play();

            timer1.Interval = 1000;
            timer1.Start();
        }

       

        private void button3_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.URL = Abrir.FileName;
            axWindowsMediaPlayer1.Ctlcontrols.pause();

            timer1.Interval = 1000;
            timer1.Start();
        }

        private void button4_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.URL = Abrir.FileName;
            axWindowsMediaPlayer1.Ctlcontrols.stop();

        }

        private void button5_Click(object sender, EventArgs e)
        {
            axWindowsMediaPlayer1.URL = Abrir.FileName;
            axWindowsMediaPlayer1.Ctlcontrols.next();

            timer1.Interval = 1000;
            timer1.Start();
        }

        private void button6_Click(object sender, EventArgs e)
        {
            Application.Exit();
        }
 
    }
