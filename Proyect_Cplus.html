//  Created by edwin reyes on 06/03/18.

#include <iostream>
using namespace std;

int main(){
    int x[5];
    int i,j,k;
    
    for(i=0;i<5;i++){
        cout<<"ingresa el valor de x["<<i+1<<"]: "; cin>>x[i];
    }
    for(i=0;i<5;i=i+1){
        for(j=0;j<5;j=j+1){
            if(x[i]<x[j]){
                k=x[i];
                x[i]=x[j];
                x[j]=k;
            }
        }
    }
    for(i=0;i<5;i++){
        cout<<x[i]<<endl;
    }
    return 0;
}

	int tf(int n);
	int ts(int n);

	int tf(int n)
	{
		int tf=0;
		int te=3;
		tf=(n-1)*(te);
		return tf;
	}
	int ts(int n)
	{
		int ts=0;
		int te=3;
		ts=(n-1)*(te)+(te);
		return ts;
	}
	int tc(int n)
	{
		int tc=0;
		int te=3;
		tc=(n)*(te);
		return tc;
	}
