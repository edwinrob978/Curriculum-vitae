Base de datos con conexion a java (registro de ventas) 

                                
create database Ventas
go
use Ventas
go
create table Productos(
idproducto int identity primary key,
descripcion varchar(80) not null,
cantidad int check(cantidad>=0),
precio float check(precio >=1.0))
go
create table Ventas(
idVenta int identity primary key,
idproducto int  foreign key references Productos(idproducto),
cantidad int check(cantidad >=0),
precio float check(precio >=1.0),
fechacompra date)
go
create procedure AltaVenta
@producto varchar(80),
@cantidad int,
@respuesta varchar(30) output
as
declare @idproducto int,@precio float
select @idproducto=idproducto,@precio=precio from Productos where descripcion=@producto
insert into Ventas(idproducto,cantidad,precio,fechacompra) values (@idproducto,@cantidad,@precio,GETDATE())
set @respuesta='Venta almacenada'
go
create trigger ActualizaExistencia
on ventas
for insert
as
declare @cantidad int
select @cantidad=cantidad from inserted
update productos set cantidad=cantidad-@cantidad
go
