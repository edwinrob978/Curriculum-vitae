package examenventas;

import java.sql.*;
import javax.swing.*;

public class Conexion {

    public static Connection conectar() {
        Connection cn = null;
        try {
            Class.forName("com.microsoft.sqlserver.jdbc.SQLServerDriver").newInstance();
            cn = DriverManager.getConnection("jdbc:sqlserver://localhost:1433;databaseName=Ventas;user=sa;password=Edwnr;");
            return (cn);

        } catch (Exception er) {
            System.out.println("Error en conectar " + er);
            return (cn);
        }
    }

    public static JComboBox llenarCombo(String query, JComboBox x) {
        Connection conexion;
        ResultSet resultado;
        x.removeAllItems();
        x.addItem("Selecciona");
        conexion = conectar();
        try {
            Statement st = conexion.createStatement();
            ResultSet rs = st.executeQuery(query);
            while (rs.next()) {
                x.addItem(rs.getString(1));
            }
            return x;
        } catch (Exception er) {
            x.removeAll();
            System.out.println("Error al llenar selector " + er);
            return (x);
        }

    }

    public static int buscarDato(String query) {
        Connection conexion;
        ResultSet resultado;
        conexion = conectar();
        try {
            Statement st = conexion.createStatement();
            ResultSet rs = st.executeQuery(query);
            if (rs.next()) {
                return (rs.getInt(1));
            }
        } catch (Exception er) {
            System.out.println("Error al buscar dato " + er);
            return (0);
        }
        return (0);
    }
}
